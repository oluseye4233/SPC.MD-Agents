handlePostPurchase(order: Order): Promise<void>; // Capability checks supportsCapability(capability: string): boolean; getAvailablePaymentMethods(): PaymentMethod[]; getMerchantOfRecordModel(): 'platform' | 'merchant'; } class GoogleUCPAdapter implements ProtocolAdapter { async discover(query: ProductQuery): Promise<Product[]> { // Call UCP Discovery API const response = await ucpClient.products.search({ query: query.text, filters: this.transformFilters(query.filters), user_profile: query.userContext }); return this.transformUCPProducts(response.products); } async initiateCheckout(cart: Cart): Promise<CheckoutSession> { // Register checkout capability with UCP const session = await ucpClient.checkout.create({ cart_id: cart.id, merchant_id: cart.merchantId, capabilities: ['direct_offers', 'loyalty_rewards'], payment_handlers: ['google_pay', 'paypal'] }); return this.transformUCPSession(session); } } class MicrosoftCopilotAdapter implements ProtocolAdapter { async discover(query: ProductQuery): Promise<Product[]> { // Analyze Copilot conversation for intent const intent = await copilotClient.conversation.analyzeIntent({ conversation_id: query.conversationId, user_message: query.text }); } } class OpenAIChatGPTAdapter implements ProtocolAdapter { async discover(query: ProductQuery): Promise<Product[]> { // Use ChatGPT Plugins API const response = await openaiClient.plugins.shopify.search({ query: query.text, filters: query.filters, user_preferences: query.userContext }); Merchant of Record Manager **Purpose:** Handles merchant-of-record variations across platforms **Platform Models:** - **Google UCP:** Retailer is merchant of record (MoR) - **Microsoft Copilot:** Flexible (platform or retailer can be MoR) - **OpenAI ChatGPT:** Platform is merchant of record **Key Responsibilities:** - Tax calculation and remittance routing - Regulatory compliance per jurisdiction - Payment settlement timing - Refund processing - Dispute resolution **Implementation:** ```typescript enum MerchantOfRecordModel { PLATFORM = 'platform', RETAILER = 'retailer', HYBRID = 'hybrid' } interface MerchantOfRecordConfig { platform: Platform; model: MerchantOfRecordModel; taxResponsibility: 'platform' | 'merchant'; paymentSettlement: 'immediate' | 'delayed'; settlementDelay?: number; // days disputeHandler: 'platform' | 'merchant'; } class MerchantOfRecordManager { async determineModel( order: Order, platform: Platform ): Promise<MerchantOfRecordConfig> { const platformConfig = this.getPlatformConfig(platform); const merchantPreferences = await this.getMerchantPreferences(order.merchantId); // Google UCP: Always retailer if (platform === 'google_ucp') { return { platform: 'google_ucp', model: MerchantOfRecordModel.RETAILER, taxResponsibility: 'merchant', paymentSettlement: 'delayed', settlementDelay: 2, disputeHandler: 'merchant' }; } // Microsoft Copilot: Flexible based on merchant preference if (platform === 'microsoft_copilot') { return merchantPreferences.preferPlatformMoR ? this.platformMoRConfig(platform) : this.merchantMoRConfig(platform); } // OpenAI ChatGPT: Always platform if (platform === 'openai_chatgpt') { return { platform: 'openai_chatgpt', model: MerchantOfRecordModel.PLATFORM, taxResponsibility: 'platform', paymentSettlement: 'immediate', disputeHandler: 'platform' }; } } async calculateTaxes( order: Order, morConfig: MerchantOfRecordConfig ): Promise<TaxCalculation> { if (morConfig.taxResponsibility === 'platform') { // Platform handles tax calculation return this.delegateToPlatform(order, morConfig.platform); } else { // Use BRIJ tax engine return this.brijTaxCalculator.calculate(order); } } } ``` ### 3. Payment Router & Processor Integration **Purpose:** Routes payments to appropriate processors based on platform and user preference **Supported Payment Methods:** - **Google UCP:** Google Pay (primary), PayPal, Card - **Microsoft Copilot:** Microsoft Pay, Card, Enterprise Procurement - **OpenAI ChatGPT:** Stripe (primary), Card, ACH **Key Responsibilities:** - Payment method selection and optimization - PCI-DSS compliant tokenization - Multi-currency support - Payment retry logic - Chargeback management **Implementation:** ```typescript interface PaymentRouter { selectOptimalPaymentMethod( user: User, platform: Platform, amount: Money ): Promise<PaymentMethod>; processPayment( payment: PaymentRequest ): Promise<PaymentResult>; handlePaymentFailure( payment: PaymentRequest, error: PaymentError ): Promise<PaymentResult>; } class BRIJPaymentRouter implements PaymentRouter { async selectOptimalPaymentMethod( user: User, platform: Platform, amount: Money ): Promise<PaymentMethod> { // Get user's saved payment methods const userMethods = await this.getUserPaymentMethods(user.id); // Get platform-supported methods const platformMethods = this.getPlatformPaymentMethods(platform); // Filter to intersection const availableMethods = userMethods.filter(m => platformMethods.includes(m.type) ); // Optimize selection based on: // 1. User preference // 2. Lowest fees // 3. Fastest processing // 4. Cashback/rewards return this.optimizeSelection(availableMethods, amount); } async processPayment(payment: PaymentRequest): Promise<PaymentResult> { // Route to appropriate processor const processor = this.getProcessor(payment.method.type); try { const result = await processor.charge({ amount: payment.amount, currency: payment.currency, source: payment.method.token, metadata: { order_id: payment.orderId, platform: payment.platform, brij_transaction_id: payment.brijId } }); // Log successful payment await this.logPaymentSuccess(payment, result); return result; } catch (error) { return this.handlePaymentFailure(payment, error); } } async handlePaymentFailure( payment: PaymentRequest, error: PaymentError ): Promise<PaymentResult> { // Categorize error if (error.type === 'insufficient_funds') { // Suggest alternative payment method return this.suggestAlternative(payment); } if (error.type === 'card_declined') { // Retry with 3DS authentication return this.retryWith3DS(payment); } if (error.type === 'network_error') { // Retry with exponential backoff return this.retryWithBackoff(payment, 3); } // Unrecoverable error return { status: 'failed', error: error, suggestions: await this.generateFailureGuidance(error) }; } } ``` --- ## 🔗 PLATFORM-SPECIFIC INTEGRATIONS ### Integration 1: Google Universal Commerce Protocol (UCP) **Protocol Version:** v1.0 **Documentation:** https://developers.google.com/commerce/ucp **Launch Date:** January 2026 #### A.
